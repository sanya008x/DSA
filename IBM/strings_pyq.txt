🔟 Strong Password
https://www.hackerrank.com/challenges/strong-password

int minimumNumber(int n, string password)
{
    int hasLower = 0, hasUpper = 0, hasDigit = 0, hasSpecial = 0;

    for(int i = 0; i < password.length(); i++)
    {
        if(password[i] >= 'a' && password[i] <= 'z')
            hasLower = 1;

        else if(password[i] >= 'A' && password[i] <= 'Z')
            hasUpper = 1;

        else if(password[i] >= '0' && password[i] <= '9')
            hasDigit = 1;

        else
            hasSpecial = 1;
    }

    int missing = 0;

    if(hasLower == 0) missing++;
    if(hasUpper == 0) missing++;
    if(hasDigit == 0) missing++;
    if(hasSpecial == 0) missing++;

    int lengthNeeded = 0;

    if(n < 6)
        lengthNeeded = 6 - n;

    // Ab decide karo final answer
    if(missing > lengthNeeded)
        return missing;
    else
        return lengthNeeded;
}

1️⃣1️⃣ Alternating Characters

int alternatingCharacters(string s) {

    int deletions = 0;

    for(int i = 0; i < s.length(); i++) {
        if(s[i] == s[i + 1]) {
            deletions++;
        }
    }

    return deletions;
}

1️⃣2️⃣ Anagram
https://www.hackerrank.com/challenges/anagram

int anagram(string s) {

    int n = s.length();

    if(n % 2 != 0)
        return -1;

    int mid = n / 2;

    string s1 = s.substr(0, mid);
    string s2 = s.substr(mid);

    int changes = 0;

    for(int i = 0; i < s1.length(); i++) {

        bool found = false;

        for(int j = 0; j < s2.length(); j++) {

            if(s1[i] == s2[j]) {
                found = true;
                s2[j] = '#';   // mark as used
                break;
            }
        }

        if(!found) {
            changes++;
        }
    }

    return changes;
}

1️⃣3️⃣ Making Anagrams
https://www.hackerrank.com/challenges/making-anagrams

int makingAnagrams(string s1, string s2) {

    int deletions = 0;

    // Step 1: Match characters of s1 in s2
    for(int i = 0; i < s1.length(); i++) {

        bool found = false;

        for(int j = 0; j < s2.length(); j++) {

            if(s1[i] == s2[j]) {
                found = true;
                s2[j] = '#';   // mark as used
                break;
            }
        }

        if(!found) {
            deletions++;   // character from s1 needs deletion
        }
    }

    // Step 2: Count remaining characters in s2
    for(int j = 0; j < s2.length(); j++) {
        if(s2[j] != '#') {
            deletions++;   // character from s2 needs deletion
        }
    }

    return deletions;
}
